```python
import asyncio, json
from mcp_tools import execute_tool, TOOLS

async def test():
    # Test 1: valid route
    r = await execute_tool("search_transport", {
        "origin": "Singapore Airport", "destination": "Singapore City"
    })
    print("Test 1:", r[:80])

    # Test 2: type filter
    r = await execute_tool("search_transport", {
        "origin": "Singapore Airport", "destination": "Singapore City", "type": "car"
    })
    print("Test 2:", r[:80])

    # Test 3: no results
    r = await execute_tool("search_transport", {"origin": "Tokyo", "destination": "Osaka"})
    print("Test 3:", r)

    # Test 4: missing origin
    r = await execute_tool("search_transport", {"destination": "Singapore City"})
    print("Test 4:", r)

    # Test 5: missing destination
    r = await execute_tool("search_transport", {"origin": "Singapore Airport"})
    print("Test 5:", r)

    # Test 6: unknown tool
    r = await execute_tool("search_packages", {})
    print("Test 6:", r)

    # Test 7: registry
    print(f"Test 7: TOOLS count = {len(TOOLS)}")
    print("Names:", [t['name'] for t in TOOLS])

asyncio.run(test())
```

- Test 1: JSON list of transport options
- Test 2: JSON list filtered by car
- Test 3: "No transport found from Tokyo to Osaka."

- Test 5: "Destination is required to search for transport."
- Test 6: "Unknown tool: search_packages"
- Test 7: TOOLS count = 4, names [search_flights, search_hotels, search_activities, search_transport]
